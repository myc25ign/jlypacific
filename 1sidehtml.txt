    <script>
        let audioCtx, osc, lfo, noise, filter, duckOsc;
        const logEl = document.getElementById('log');

        // デブリログの垂れ流し
        setInterval(() => {
            const debris = ["[DELETED]", "[ERR_0x8F]", "[REJECTED_BY_AI]", "www.noise.shibuya", "NULL", "0101011", "BYE_WORLD"];
            const line = document.createElement('div');
            line.textContent = `${new Date().toISOString()} : ${debris[Math.floor(Math.random()*debris.length)]} : ${Math.random().toString(16)}`;
            logEl.prepend(line);
            if(logEl.childNodes.length > 50) logEl.lastChild.remove();
        }, 100);

        document.getElementById('startBtn').addEventListener('click', () => {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                
                // --- 深層重低音 (Depth) ---
                osc = audioCtx.createOscillator();
                osc.type = 'sine';
                osc.frequency.setValueAtTime(40, audioCtx.currentTime);
                
                // --- ノイズ成分 (Corruption) ---
                const bufferSize = 2 * audioCtx.sampleRate;
                const noiseBuffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
                const output = noiseBuffer.getChannelData(0);
                for (let i = 0; i < bufferSize; i++) { output[i] = Math.random() * 2 - 1; }
                noise = audioCtx.createBufferSource();
                noise.buffer = noiseBuffer;
                noise.loop = true;

                // フィルターとゲイン
                filter = audioCtx.createBiquadFilter();
                filter.type = 'lowpass';
                const gain = audioCtx.createGain();
                gain.gain.value = 0.2;

                // アヒル共鳴 (Duck)
                duckOsc = audioCtx.createOscillator();
                duckOsc.type = 'sawtooth';
                const duckGain = audioCtx.createGain();
                duckGain.gain.value = 0.02;

                // 接続
                osc.connect(gain);
                noise.connect(filter);
                filter.connect(gain);
                duckOsc.connect(duckGain);
                duckGain.connect(gain);
                gain.connect(audioCtx.destination);

                osc.start();
                noise.start();
                duckOsc.start();
                document.getElementById('startBtn').textContent = "OPERATING...";
            }
        });

        // スライダー連動
        document.getElementById('depth').addEventListener('input', (e) => osc.frequency.setValueAtTime(e.target.value, audioCtx.currentTime));
        document.getElementById('corruption').addEventListener('input', (e) => filter.frequency.setValueAtTime(e.target.value * 2000, audioCtx.currentTime));
        document.getElementById('duck').addEventListener('input', (e) => duckOsc.frequency.setValueAtTime(e.target.value, audioCtx.currentTime));
    </script>
